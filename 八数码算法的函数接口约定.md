```python
'''
状态节点类
'''
class Node:
    '''
    data表示状态的字符串,由数字1到8以及空格组成, 
    如'123 45678',表示的棋盘如下:
    1 2 3
      4 5
    6 7 8
    cost表示从初始状态结点到此结点的耗费,取值等于结点的深度
    parent表示结点的父指针
    '''
    def __init__(self, data, cost = 0, parent = None):
        self.data = data
        self.parent = parent# 指向父亲结点
        self.cost = cost
        

```

```python
'''
包含五个搜索算法的八数码求解器类
'''
class Solver:
    def __init__(self, start, target='1238 4765', max_depth=3):
        self.start = start#初始状态
        self.target = target#目标状态
        self.max_depth = max_depth#搜索的最大深度
        '''
        self.G保存搜索图,用于查找解路 
        字典类型,保存状态结点, key=状态字符串, value=状态结点
        例如 G= {'1238 4765': Node()}表示搜索图中包含一个状态为'1238 4765'的结点,它的值是节点类型
        '''
        self.G = {}
```
